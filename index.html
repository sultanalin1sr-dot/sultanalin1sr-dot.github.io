// frontend-chat.js const BACKEND_URL = 'https://YOUR-PROJECT.vercel.app/api/chat'; const history = []; async function sendMessage(message) { const res = await fetch(BACKEND_URL, { method: 'POST', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify({ message, history }), }); if (!res.ok) { const errBody = await res.text(); console.error('Chat backend error:', res.status, errBody); throw new Error(`Backend returned ${res.status}`); } const data = await res.json(); if (!data.reply) { console.error('No reply:', data); throw new Error('Empty reply from backend'); } history.push({ role: 'user', content: message }); history.push({ role: 'assistant', content: data.reply }); return data.reply; } document.addEventListener('DOMContentLoaded', () => { const form = document.getElementById('chat-form'); const input = document.getElementById('chat-input'); const log = document.getElementById('chat-log'); if (!form || !input || !log) return; form.addEventListener('submit', async (e) => { e.preventDefault(); const text = input.value.trim(); if (!text) return; input.value = ''; appendLine(log, 'You', text); try { const reply = await sendMessage(text); appendLine(log, 'Bot', reply); } catch (err) { appendLine(log, 'Bot', `[Error: ${err.message}]`); } }); }); function appendLine(log, who, text) { const div = document.createElement('div'); div.textContent = `${who}: ${text}`; log.appendChild(div); log.scrollTop = log.scrollHeight; }
